function conectar() {
    $server = 'localhost';
    $bd = 'empresa';
    $user = 'root';
    $pass = '';
    $con = mysqli_connect($server, $user, $pass, $bd);

    // Verificar si hay errores en la conexiÃ³n
    if (mysqli_connect_errno()) {
        die("Error al conectar a la base de datos: " . mysqli_connect_error());
    }

    return $con;
}
